** List Processing
~coerce~ data to/from lists, strings, etc for easy processing

#+BEGIN_SRC lisp
(defun game-print (lst)
  (princ (coerce (tweak-text (coerce (string-trim "() "
						  (prin1-to-string lst))
				     'list)
			     t
			     nil)
		 'string))
  (fresh-line))
#+END_SRC
